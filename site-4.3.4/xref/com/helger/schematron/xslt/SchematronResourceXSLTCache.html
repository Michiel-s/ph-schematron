32  import com.helger.commons.collection.ext.CommonsHashMap;
33  import com.helger.commons.collection.ext.ICommonsMap;
34  import com.helger.commons.concurrent.SimpleReadWriteLock;
35  import com.helger.commons.error.IError;
36  import com.helger.commons.io.resource.IReadableResource;
37  import com.helger.xml.transform.CollectingTransformErrorListener;
38  import com.helger.xml.transform.LoggingTransformErrorListener;

40  /**
41   * Factory for creating {@link ISchematronXSLTBasedProvider} objects.
42   *
43   * @author Philip Helger
44   */
45  @ThreadSafe
46  public final class SchematronResourceXSLTCache
47  {
48    private static final Logger s_aLogger = LoggerFactory.getLogger (SchematronResourceXSLTCache.class);
49    private static final SimpleReadWriteLock s_aRWLock = new SimpleReadWriteLock ();
50    @GuardedBy ("s_aRWLock")
51    private static final ICommonsMap <String, SchematronProviderXSLTPrebuild> s_aCache = new CommonsHashMap <> ();
52  
53    private SchematronResourceXSLTCache ()
54    {}
55  
56    @Nullable
57    public static SchematronProviderXSLTPrebuild createSchematronXSLTProvider (@Nonnull final IReadableResource aXSLTResource,
58                                                                               @Nullable final ErrorListener aCustomErrorListener,
59                                                                               @Nullable final URIResolver aCustomURIResolver)
60    {
61      if (s_aLogger.isInfoEnabled ())
62        s_aLogger.info ("Compiling XSLT instance " + aXSLTResource.toString ());
63  
64      final CollectingTransformErrorListener aCEH = new CollectingTransformErrorListener ();
65      final SchematronProviderXSLTPrebuild aXSLTPreprocessor = new SchematronProviderXSLTPrebuild (aXSLTResource,
66                                                                                                   aCEH.andThen (aCustomErrorListener != null ? aCustomErrorListener
67                                                                                                                                              : new LoggingTransformErrorListener (Locale.US)),
68                                                                                                   aCustomURIResolver);
69      if (!aXSLTPreprocessor.isValidSchematron ())
70      {
71        // Schematron is invalid -> parsing failed
72        s_aLogger.warn ("The XSLT resource '" + aXSLTResource.getResourceID () + "' is invalid!");
73        for (final IError aError : aCEH.getErrorList ())
74          s_aLogger.warn ("  " + aError.getAsString (Locale.US));
75        return null;
76      }
77  
78      // If it is a valid schematron, there must be a result XSLT present!
79      if (aXSLTPreprocessor.getXSLTDocument () == null)
80      {
81        // Note: this should never occur, as it is in the Prebuild implementation
82        // the same as "isValidSchematron" but to be implementation agnostic, we
83        // leave the check anyway.
84        throw new IllegalStateException ("No XSLT document retrieved from XSLT resource '" +
85                                         aXSLTResource.getResourceID () +
86                                         "'!");
87      }
88  
89      // Create the main validator for the schematron
90      return aXSLTPreprocessor;
91    }
92  
93    /**
94     * Return an existing or create a new Schematron XSLT provider for the passed
95     * resource.
96     *
97     * @param aXSLTResource
98     *        The resource of the Schematron rules. May not be <code>null</code>.
99     * @param aCustomErrorListener
100    *        The custom error listener to be used. May be <code>null</code>.
101    * @param aCustomURIResolver
102    *        The custom URI resolver to be used. May be <code>null</code>.
103    * @return <code>null</code> if the passed Schematron XSLT resource does not
104    *         exist.
105    */
106   @Nullable
107   public static SchematronProviderXSLTPrebuild getSchematronXSLTProvider (@Nonnull final IReadableResource aXSLTResource,
108                                                                           @Nullable final ErrorListener aCustomErrorListener,
109                                                                           @Nullable final URIResolver aCustomURIResolver)
110   {
111     ValueEnforcer.notNull (aXSLTResource, "resource");
112 
113     if (!aXSLTResource.exists ())
114     {
115       s_aLogger.warn ("XSLT resource " + aXSLTResource + " does not exist!");
116       return null;
117     }
118 
119     // Determine the unique resource ID for caching
120     final String sResourceID = aXSLTResource.getResourceID ();
121 
122     // Validator already in the cache?
123     final SchematronProviderXSLTPrebuild aProvider = s_aRWLock.readLocked ( () -> s_aCache.get (sResourceID));
124     if (aProvider != null)
125       return aProvider;
126 
127     return s_aRWLock.writeLocked ( () -> {
128       // Check again in write lock
129       SchematronProviderXSLTPrebuild aProvider2 = s_aCache.get (sResourceID);
130       if (aProvider2 == null)
131       {
132         // Create new object and put in cache
133         aProvider2 = createSchematronXSLTProvider (aXSLTResource, aCustomErrorListener, aCustomURIResolver);
134         if (aProvider2 != null)
135           s_aCache.put (sResourceID, aProvider2);
136       }
137       return aProvider2;
138     });
139   }

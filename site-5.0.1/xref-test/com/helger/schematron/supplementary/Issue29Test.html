42  /**
43   * Test code for issue #29
44   *
45   * @author Philip Helger
46   */
47  public final class Issue29Test
48  {
49    private static final Logger s_aLogger = LoggerFactory.getLogger (Issue29Test.class);
50  
51    @Nullable
52    static SchematronOutputType validateXmlUsingSchematron (@Nonnull final IReadableResource aRes)
53    {
54      SchematronOutputType ob = null;
55  
56      // Must use the XSLT based version, because of "key" usage
57      final ISchematronResource aResSCH = new SchematronResourcePure (new ClassPathResource ("issues/github29/pbs.sch"));
58      if (!aResSCH.isValidSchematron ())
59        throw new IllegalArgumentException ("Invalid Schematron!");
60      try
61      {
62        final Document aDoc = aResSCH.applySchematronValidation (new ResourceStreamSource (aRes));
63        if (aDoc != null)
64        {
65          final SVRLMarshaller marshaller = new SVRLMarshaller ();
66          ob = marshaller.read (aDoc);
67        }
68      }
69      catch (final Exception pE)
70      {
71        pE.printStackTrace ();
72      }
73      return ob;
74    }
75  
76    @Test
77    @Ignore ("Takes too long - more than 1 min")
78    public void testGood () throws Exception
79    {
80      final SchematronOutputType aSOT = validateXmlUsingSchematron (new GZIPReadableResource (new ClassPathResource ("issues/github29/sample.xml.gz")));
81      assertNotNull (aSOT);
82      final ICommonsList <SVRLFailedAssert> aErrors = SVRLHelper.getAllFailedAssertions (aSOT);
83      assertNotNull (aErrors);
84      s_aLogger.info ("Errors found: " + aErrors);
85    }
86  }

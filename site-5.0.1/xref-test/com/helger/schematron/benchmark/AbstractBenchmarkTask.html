17  package com.helger.schematron.benchmark;
18  
19  import java.util.Date;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import com.helger.commons.CGlobal;
25  import com.helger.commons.system.SystemHelper;
26  import com.helger.commons.system.SystemProperties;
27  
28  /**
29   * Abstract benchmarking class.
30   *
31   * @author Philip Helger
32   */
33  public abstract class AbstractBenchmarkTask
34  {
35    /** logger to use. */
36    protected static final Logger s_aLogger = LoggerFactory.getLogger (AbstractBenchmarkTask.class);
37  
38    /**
39     * This constant specifies the minimum number of times that a task to be
40     * benchmarked must run before an actual timing measurement should be done.
41     * This is done both to load all relevant classes as well as to warm up
42     * hotspot before doing a measurement.
43     */
44    private static final int MIN_WARMUP_CALLS = 1000;
45    /**
46     * This constant specifies the minimum time that a task to be benchmarked must
47     * run to assure accurate timing measurements. It must be set reasonably large
48     * to deal with the low resolution clocks present on many platforms (e.g.
49     * typical Windows machines have error around 10s of ms?). The units of this
50     * constant are nanoseconds.
51     */
52    private static final long MIN_BENCHMARK_NANOSECS = 3 * CGlobal.NANOSECONDS_PER_SECOND;
53  
54    protected AbstractBenchmarkTask ()
55    {}
56  
57    protected static void logSystemInfo ()
58    {
59      s_aLogger.info ("Runtime: Date=" +
60                      new Date ().toString () +
61                      "; Java=" +
62                      SystemProperties.getJavaVersion () +
63                      "; OS=" +
64                      SystemHelper.getOperatingSystemName () +
65                      "; User=" +
66                      SystemProperties.getUserName () +
67                      "; Procs=" +
68                      SystemHelper.getNumberOfProcessors ());
69    }
70  
71    /**
72     * Measures how long it takes to execute the run method of the task arg.
73     * <p>
74     * This method uses the MIN_WARMUP_CALLS and MIN_BENCHMARK_TIME constants to
75     * obtain accurate results.
76     * <p>
77     * This method explicitly requests garbage collection before doing each
78     * benchmark. Therefore, unless you actually want to include the effect of
79     * garbage collection in the benchmark, the JVM should use a "stop-the-world"
80     * garbage collector, if it is available. (Usually it is. With Sun's tools,
81     * this is, in fact, the default garbage collector type.) The type of garbage
82     * collectors to avoid are incremental, concurrent, or parallel ones.
83     * <p>
84     *
85     * @param aTask
86     *        The task to be executed
87     * @return average execution time of a single invocation of task.run, in
88     *         microseconds
89     */
90    protected static final double benchmarkTask (final Runnable aTask)
91    {
92      int numberOfRuns = MIN_WARMUP_CALLS;
93      while (true)
94      {
95        System.gc ();
96        final long t1 = System.nanoTime ();
97        for (int i = 0; i < numberOfRuns; i++)
98        {
99          aTask.run ();
100       }
101       final long t2 = System.nanoTime ();
102 
103       final long nTestNanos = t2 - t1;
104       if ((numberOfRuns > MIN_WARMUP_CALLS) && (nTestNanos > MIN_BENCHMARK_NANOSECS))
105       {
106         // System.out.println("numberOfRuns required = " + numberOfRuns);
107         return nTestNanos / (double) numberOfRuns;
108       }
109       numberOfRuns *= 2;
110     }
111   }
112 }

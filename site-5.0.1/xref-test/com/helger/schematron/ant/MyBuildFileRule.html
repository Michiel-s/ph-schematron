33  /**
34   * Provides access for JUnit tests to execute Ant targets and access execution
35   * details (i.e logs). Example usage: <code>
36   * public class MyTest {
37   *
38   *     \@Rule
39   *     public BuildFileRule rule = new BuildFileRule();
40   *
41   *     \@Before
42   *     public void setUp() {
43   *         rule.configureProject("my/and/file.xml");
44   *     }
45   *
46   *     \@Test
47   *     public void testSuccess() {
48   *         rule.executeTarget("passingTaget");
49   *         assertEquals("Incorrect log message", "[taskName] Action Complete", rule.getLog());
50   *     }
51   *
52   *     \@Test
53   *     public void testException() {
54   *         try {
55   *             rule.executeTarget("failingTarget");
56   *             fail("Target should have thrown a BuildException");
57   *         } catch (BuildException ex) {
58   *             assertContains("Exception did not contain correct text", "Could not find compiler on classpath", ex.getMessage());
59   *         }
60   *     }
61   *
62   * }
63   * </code>
64   */
65  public final class MyBuildFileRule extends ExternalResource
66  {
67    private Project project;
68    private StringBuffer logBuffer;
69    private StringBuffer fullLogBuffer;
70    private StringBuffer outputBuffer;
71    private StringBuffer errorBuffer;
72  
73    /**
74     * Tidies up following a test execution. If the currently configured project
75     * has a <tt>tearDown</tt> target then this will automatically be called,
76     * otherwise this method will not perform any actions.
77     */
78    @Override
79    protected void after ()
80    {
81      if (project == null)
82      {
83        // configureProject has not been called - nothing we can clean-up
84        return;
85      }
86      final String tearDown = "tearDown";
87      if (project.getTargets ().containsKey (tearDown))
88      {
89        project.executeTarget (tearDown);
90      }
91    }
92  
93    /**
94     * Gets the INFO, WARNING and ERROR message from the current execution, unless
95     * the logging level is set above any of these level in which case the message
96     * is excluded. This is only valid if configureProject() has been called.
97     *
98     * @return The INFO, WARN and ERROR messages in the log.
99     */
100   public String getLog ()
101   {
102     return logBuffer.toString ();
103   }
104 
105   /**
106    * Gets any messages that have been logged during the current execution,
107    * unless the logging level has been set above the log level defined in the
108    * message. Only valid if configureProject() has been called.
109    *
110    * @return the content of the log.
111    */
112   public String getFullLog ()
113   {
114     return fullLogBuffer.toString ();
115   }
116 
117   /**
118    * Provides all output sent to the System.out stream during the current
119    * execution.
120    *
121    * @return all output messages in a single string, normalised to have platform
122    *         independent line breaks.
123    */
124   public String getOutput ()
125   {
126     return _cleanBuffer (outputBuffer);
127   }
128 
129   /**
130    * Provides all output sent to the System.err stream during the current
131    * execution.
132    *
133    * @return all error messages in a single string, normalised to have platform
134    *         independent line breaks.
135    */
136   public String getError ()
137   {
138     return _cleanBuffer (errorBuffer);
139   }
140 
141   private String _cleanBuffer (final StringBuffer buffer)
142   {
143     final StringBuilder cleanedBuffer = new StringBuilder (buffer.length ());
144     for (int i = 0; i < buffer.length (); i++)
145     {
146       final char ch = buffer.charAt (i);
147       if (ch != '\r')
148         cleanedBuffer.append (ch);
149     }
150     return cleanedBuffer.toString ();
151   }

153   /**
154    * Sets up to run the named project
155    *
156    * @param filename
157    *        name of project file to run
158    * @throws BuildException
159    *         on error
160    */
161   public void configureProject (final String filename) throws BuildException
162   {
163     configureProject (filename, Project.MSG_DEBUG);
164   }
165 
166   /**
167    * Sets up to run the named project
168    *
169    * @param filename
170    *        name of project file to run
171    * @param logLevel
172    *        Log level
173    * @throws BuildException
174    *         on error
175    */
176   public void configureProject (final String filename, final int logLevel) throws BuildException
177   {
178     logBuffer = new StringBuffer ();
179     fullLogBuffer = new StringBuffer ();
180     project = new Project ();
181     project.init ();
182     final File antFile = new File (System.getProperty ("root"), filename);
183     project.setProperty ("ant.processid", ProcessUtil.getProcessId ("<Process>"));
184     project.setProperty ("ant.threadname", Thread.currentThread ().getName ());
185     project.setUserProperty ("ant.file", antFile.getAbsolutePath ());
186     project.addBuildListener (new AntTestListener (logLevel));
187     ProjectHelper.configureProject (project, antFile);
188   }
189 
190   /**
191    * Executes a target in the configured Ant build file. Requires
192    * #configureProject() to have been invoked before this call.
193    *
194    * @param targetName
195    *        the target in the currently configured build file to run.
196    */
197   public void executeTarget (final String targetName)
198   {
199     outputBuffer = new StringBuffer ();
200     final PrintStream out = new PrintStream (new AntOutputStream (outputBuffer));
201     errorBuffer = new StringBuffer ();
202     final PrintStream err = new PrintStream (new AntOutputStream (errorBuffer));
203     logBuffer = new StringBuffer ();
204     fullLogBuffer = new StringBuffer ();
205 
206     /*
207      * we synchronize to protect our custom output streams from being overridden
208      * by other tests executing targets concurrently. Ultimately this would only
209      * happen if we ran a multi-threaded test executing multiple targets at
210      * once, and this protection doesn't prevent a target from internally
211      * modifying the output stream during a test - but at least this scenario is
212      * fairly deterministic so easier to troubleshoot.
213      */
214     synchronized (System.out)
215     {
216       final PrintStream sysOut = System.out;
217       final PrintStream sysErr = System.err;
218       sysOut.flush ();
219       sysErr.flush ();
220       try
221       {
222         System.setOut (out);
223         System.setErr (err);
224         project.executeTarget (targetName);
225       }
226       finally
227       {
228         System.setOut (sysOut);
229         System.setErr (sysErr);
230       }
231     }
232   }

244   /**
245    * An output stream which saves contents to our buffer.
246    */
247   protected static class AntOutputStream extends OutputStream
248   {
249     private final StringBuffer m_aSB;
250 
251     public AntOutputStream (final StringBuffer buffer)
252     {
253       m_aSB = buffer;
254     }
255 
256     @Override
257     public void write (final int b)
258     {
259       m_aSB.append ((char) b);
260     }
261   }

263   /**
264    * Our own personal build listener.
265    */
266   private class AntTestListener implements BuildListener
267   {
268     private final int m_nLogLevel;
269 
270     /**
271      * Constructs a test listener which will ignore log events above the given
272      * level.
273      *
274      * @param logLevel
275      *        Log level
276      */
277     public AntTestListener (final int logLevel)
278     {
279       m_nLogLevel = logLevel;
280     }
281 
282     /**
283      * Fired before any targets are started.
284      */
285     public void buildStarted (final BuildEvent event)
286     {}
287 
288     /**
289      * Fired after the last target has finished. This event will still be thrown
290      * if an error occurred during the build.
291      *
292      * @see BuildEvent#getException()
293      */
294     public void buildFinished (final BuildEvent event)
295     {}
296 
297     /**
298      * Fired when a target is started.
299      *
300      * @see BuildEvent#getTarget()
301      */
302     public void targetStarted (final BuildEvent event)
303     {}
304 
305     /**
306      * Fired when a target has finished. This event will still be thrown if an
307      * error occurred during the build.
308      *
309      * @see BuildEvent#getException()
310      */
311     public void targetFinished (final BuildEvent event)
312     {}
313 
314     /**
315      * Fired when a task is started.
316      *
317      * @see BuildEvent#getTask()
318      */
319     public void taskStarted (final BuildEvent event)
320     {}
321 
322     /**
323      * Fired when a task has finished. This event will still be throw if an
324      * error occurred during the build.
325      *
326      * @see BuildEvent#getException()
327      */
328     public void taskFinished (final BuildEvent event)
329     {}
330 
331     /**
332      * Fired whenever a message is logged.
333      *
334      * @see BuildEvent#getMessage()
335      * @see BuildEvent#getPriority()
336      */
337     public void messageLogged (final BuildEvent event)
338     {
339       if (event.getPriority () > m_nLogLevel)
340       {
341         // ignore event
342         return;
343       }
344 
345       if (event.getPriority () == Project.MSG_INFO ||
346           event.getPriority () == Project.MSG_WARN ||
347           event.getPriority () == Project.MSG_ERR)
348       {
349         logBuffer.append (event.getMessage ()).append ('\n');
350       }
351       fullLogBuffer.append (event.getMessage ()).append ('\n');
352     }
353   }

355   @Nonnull
356   public File getOutputDir ()
357   {
358     return new File (getProject ().getProperty ("output"));

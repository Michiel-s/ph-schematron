53  /**
54   * Converts one or more Schematron schema files into XSLT scripts.
55   *
56   * @goal convert
57   * @phase generate-resources
58   * @author PEPPOL.AT, BRZ, Philip Helger
59   */
60  @SuppressFBWarnings ({ "NP_UNWRITTEN_FIELD", "UWF_UNWRITTEN_FIELD" })
61  public final class Schematron2XSLTMojo extends AbstractMojo
62  {
63    public final class PluginErrorListener extends AbstractTransformErrorListener
64    {
65      private final File m_aSourceFile;
66  
67      public PluginErrorListener (@Nonnull final File aSource)
68      {
69        m_aSourceFile = aSource;
70      }
71  
72      @Override
73      protected void internalLog (@Nonnull final IError aResError)
74      {
75        final int nLine = aResError.getErrorLocation ().getLineNumber ();
76        final int nColumn = aResError.getErrorLocation ().getColumnNumber ();
77        final String sMessage = StringHelper.getImplodedNonEmpty (" - ",
78                                                                  aResError.getErrorText (Locale.US),
79                                                                  aResError.getLinkedExceptionMessage ());
80  
81        // 0 means undefined line/column
82        buildContext.addMessage (m_aSourceFile,
83                                 nLine <= 0 ? 0 : nLine,
84                                 nColumn <= 0 ? 0 : nColumn,
85                                 sMessage,
86                                 aResError.isError () ? BuildContext.SEVERITY_ERROR : BuildContext.SEVERITY_WARNING,
87                                 aResError.getLinkedExceptionCause ());
88      }
89    }

109   /**

117   /**
118    * A pattern for the Schematron files. Can contain Ant-style wildcards and
119    * double wildcards. All files that match the pattern will be converted. Files
120    * in the schematronDirectory and its subdirectories will be considered.
121    *
122    * @parameter property="schematronPattern" default-value="**\/*.sch"

124   private String schematronPattern;

126   /**
127    * The directory where the XSLT files will be saved.

129    * @required
130    * @parameter property="xsltDirectory"
131    *            default-value="${basedir}/src/main/xslt"

133   private File xsltDirectory;
134 
135   /**
136    * The file extension of the created XSLT files.
137    *
138    * @parameter property="xsltExtension" default-value=".xslt"
139    */
140   private String xsltExtension;
141 
142   /**
143    * Overwrite existing Schematron files without notice? If this is set to
144    * <code>false</code> than existing XSLT files are not overwritten.
145    *
146    * @parameter property="overwrite" default-value="true"
147    */
148   private boolean overwriteWithoutQuestion = true;
149 
150   /**
151    * Define the phase to be used for XSLT creation. By default the
152    * <code>defaultPhase</code> attribute of the Schematron file is used.

154    * @parameter property="phaseName"
155    */
156   private String phaseName;
157 
158   /**
159    * Define the language code for the XSLT creation. Default is English.
160    * Supported language codes are: cs, de, en, fr, nl.
161    *
162    * @parameter property="languageCode"
163    */
164   private String languageCode;

166   public void setSchematronDirectory (@Nonnull final File aDir)
167   {
168     schematronDirectory = aDir;
169     if (!schematronDirectory.isAbsolute ())
170       schematronDirectory = new File (project.getBasedir (), aDir.getPath ());
171     getLog ().debug ("Searching Schematron files in the directory '" + schematronDirectory + "'");
172   }
173 
174   public void setSchematronPattern (final String sPattern)
175   {
176     schematronPattern = sPattern;
177     getLog ().debug ("Setting Schematron pattern to '" + sPattern + "'");
178   }
179 
180   public void setXsltDirectory (@Nonnull final File aDir)
181   {
182     xsltDirectory = aDir;
183     if (!xsltDirectory.isAbsolute ())
184       xsltDirectory = new File (project.getBasedir (), aDir.getPath ());
185     getLog ().debug ("Writing XSLT files into directory '" + xsltDirectory + "'");
186   }
187 
188   public void setXsltExtension (final String sExt)
189   {
190     xsltExtension = sExt;
191     getLog ().debug ("Setting XSLT file extension to '" + sExt + "'");
192   }
193 
194   public void setOverwriteWithoutQuestion (final boolean bOverwrite)
195   {
196     overwriteWithoutQuestion = bOverwrite;
197     if (overwriteWithoutQuestion)
198       getLog ().debug ("Overwriting XSLT files without notice");
199     else
200       getLog ().debug ("Ignoring existing Schematron files");
201   }
202 
203   public void setPhaseName (final String sPhaseName)
204   {
205     phaseName = sPhaseName;
206     if (phaseName == null)
207       getLog ().debug ("Using default phase");
208     else
209       getLog ().debug ("Using the phase '" + phaseName + "'");
210   }
211 
212   public void setLanguageCode (final String sLanguageCode)
213   {
214     languageCode = sLanguageCode;
215     if (languageCode == null)
216       getLog ().debug ("Using default language code");
217     else
218       getLog ().debug ("Using the language code '" + languageCode + "'");
219   }

221   public void execute () throws MojoExecutionException, MojoFailureException
222   {
223     StaticLoggerBinder.getSingleton ().setMavenLog (getLog ());
224     if (schematronDirectory == null)
225       throw new MojoExecutionException ("No Schematron directory specified!");
226     if (schematronDirectory.exists () && !schematronDirectory.isDirectory ())
227       throw new MojoExecutionException ("The specified Schematron directory " +
228                                         schematronDirectory +
229                                         " is not a directory!");
230     if (StringHelper.hasNoText (schematronPattern))
231       throw new MojoExecutionException ("No Schematron pattern specified!");
232     if (xsltDirectory == null)
233       throw new MojoExecutionException ("No XSLT directory specified!");
234     if (xsltDirectory.exists () && !xsltDirectory.isDirectory ())
235       throw new MojoExecutionException ("The specified XSLT directory " + xsltDirectory + " is not a directory!");
236     if (StringHelper.hasNoText (xsltExtension) || !xsltExtension.startsWith ("."))
237       throw new MojoExecutionException ("The XSLT extension '" + xsltExtension + "' is invalid!");
238 
239     if (!xsltDirectory.exists () && !xsltDirectory.mkdirs ())
240       throw new MojoExecutionException ("Failed to create the XSLT directory " + xsltDirectory);
241 
242     // for all Schematron files that match the pattern
243     final DirectoryScanner aScanner = new DirectoryScanner ();
244     aScanner.setBasedir (schematronDirectory);
245     aScanner.setIncludes (new String [] { schematronPattern });
246     aScanner.setCaseSensitive (true);
247     aScanner.scan ();
248     final String [] aFilenames = aScanner.getIncludedFiles ();
249     if (aFilenames != null)
250     {
251       for (final String sFilename : aFilenames)
252       {
253         final File aFile = new File (schematronDirectory, sFilename);
254 
255         // 1. build XSLT file name (outputdir + localpath with new extension)
256         final File aXSLTFile = new File (xsltDirectory, FilenameHelper.getWithoutExtension (sFilename) + xsltExtension);
257 
258         getLog ().info ("Converting Schematron file '" +
259                         aFile.getPath () +
260                         "' to XSLT file '" +
261                         aXSLTFile.getPath () +
262                         "'");
263 
264         // 2. The Schematron resource
265         final IReadableResource aSchematronResource = new FileSystemResource (aFile);
266 
267         // 3. Check if the XSLT file already exists
268         if (aXSLTFile.exists () && !overwriteWithoutQuestion)
269         {
270           // 3.1 Not overwriting the existing file
271           getLog ().debug ("Skipping XSLT file '" + aXSLTFile.getPath () + "' because it already exists!");
272         }
273         else
274         {
275           // 3.2 Create the directory, if necessary
276           final File aXsltFileDirectory = aXSLTFile.getParentFile ();
277           if (aXsltFileDirectory != null && !aXsltFileDirectory.exists ())
278           {
279             getLog ().debug ("Creating directory '" + aXsltFileDirectory.getPath () + "'");
280             if (!aXsltFileDirectory.mkdirs ())
281             {
282               final String message = "Failed to convert '" +
283                                      aFile.getPath () +
284                                      "' because directory '" +
285                                      aXsltFileDirectory.getPath () +
286                                      "' could not be created";
287               getLog ().error (message);
288               throw new MojoFailureException (message);
289             }
290           }
291           // 3.3 Okay, write the XSLT file
292           try
293           {
294             buildContext.removeMessages (aFile);
295             // Custom error listener to log to the Mojo logger
296             final ErrorListener aMojoErrorListener = new PluginErrorListener (aFile);
297 
298             // Custom error listener
299             // No custom URI resolver
300             // Specified phase - default = null
301             // Specified language code - default = null
302             final ISchematronXSLTBasedProvider aXsltProvider = SchematronResourceSCHCache.createSchematronXSLTProvider (aSchematronResource,
303                                                                                                                         new SCHTransformerCustomizer ().setErrorListener (aMojoErrorListener)
304                                                                                                                                                        .setPhase (phaseName)
305                                                                                                                                                        .setLanguageCode (languageCode));
306             if (aXsltProvider != null)
307             {
308               // Write the resulting XSLT file to disk
309               final MapBasedNamespaceContext aNSContext = new MapBasedNamespaceContext ().addMapping ("svrl",
310                                                                                                       CSVRL.SVRL_NAMESPACE_URI);
311               // Add all namespaces from XSLT document root
312               final String sNSPrefix = XMLConstants.XMLNS_ATTRIBUTE + ":";
313               XMLHelper.forAllAttributes (aXsltProvider.getXSLTDocument ().getDocumentElement (),
314                                           (sAttrName, sAttrValue) -> {
315                                             if (sAttrName.startsWith (sNSPrefix))
316                                               aNSContext.addMapping (sAttrName.substring (sNSPrefix.length ()),
317                                                                      sAttrValue);
318                                           });
319 
320               final XMLWriterSettings aXWS = new XMLWriterSettings ();
321               aXWS.setNamespaceContext (aNSContext).setPutNamespaceContextPrefixesInRoot (true);
322 
323               final OutputStream aOS = FileHelper.getOutputStream (aXSLTFile);
324               if (aOS == null)
325                 throw new IllegalStateException ("Failed to open output stream for file " +
326                                                  aXSLTFile.getAbsolutePath ());
327               XMLWriter.writeToStream (aXsltProvider.getXSLTDocument (), aOS, aXWS);
328               buildContext.refresh (aXsltFileDirectory);
329             }
330             else
331             {
332               final String message = "Failed to convert '" + aFile.getPath () + "': the Schematron resource is invalid";
333               getLog ().error (message);
334               throw new MojoFailureException (message);
335             }
336           }
337           catch (final MojoFailureException up)
338           {
339             throw up;
340           }
341           catch (final Exception ex)
342           {
343             final String sMessage = "Failed to convert '" +
344                                     aFile.getPath () +
345                                     "' to XSLT file '" +
346                                     aXSLTFile.getPath () +
347                                     "'";
348             getLog ().error (sMessage, ex);
349             throw new MojoExecutionException (sMessage, ex);
350           }
351         }
352       }
353     }
354   }

55  /**
56   * Applies Schematron validation onto an XML file
57   *
58   * @goal validate
59   * @phase process-resources
60   * @author Philip Helger
61   */
62  @SuppressFBWarnings ({ "NP_UNWRITTEN_FIELD", "UWF_UNWRITTEN_FIELD" })
63  public final class SchematronValidationMojo extends AbstractMojo

65    public final class PluginErrorListener extends AbstractTransformErrorListener
66    {
67      private final File m_aSourceFile;
68  
69      public PluginErrorListener (@Nonnull final File aSource)
70      {
71        m_aSourceFile = aSource;
72      }
73  
74      @Override
75      protected void internalLog (@Nonnull final IError aResError)
76      {
77        final int nLine = aResError.getErrorLocation ().getLineNumber ();
78        final int nColumn = aResError.getErrorLocation ().getColumnNumber ();
79        final String sMessage = StringHelper.getImplodedNonEmpty (" - ",
80                                                                  aResError.getErrorText (Locale.US),
81                                                                  aResError.getLinkedExceptionMessage ());
82  
83        // 0 means undefined line/column
84        buildContext.addMessage (m_aSourceFile,
85                                 nLine <= 0 ? 0 : nLine,
86                                 nColumn <= 0 ? 0 : nColumn,
87                                 sMessage,
88                                 aResError.isError () ? BuildContext.SEVERITY_ERROR : BuildContext.SEVERITY_WARNING,
89                                 aResError.getLinkedExceptionCause ());
90      }
91    }

322   private void _performValidation (@Nonnull final ISchematronResource aSch,
323                                    @Nonnull final File aXMLDirectory,
324                                    @Nullable final String sXMLIncludes,
325                                    @Nullable final String sXMLExcludes,
326                                    @Nullable final File aSVRLDirectory,
327                                    final boolean bExpectSuccess) throws MojoExecutionException, MojoFailureException
328   {
329     final DirectoryScanner aScanner = new DirectoryScanner ();
330     aScanner.setBasedir (aXMLDirectory);
331     if (StringHelper.hasText (sXMLIncludes))
332       aScanner.setIncludes (new String [] { sXMLIncludes });
333     if (StringHelper.hasText (sXMLExcludes))
334       aScanner.setExcludes (new String [] { sXMLExcludes });
335     aScanner.setCaseSensitive (true);
336     aScanner.scan ();
337     final String [] aXMLFilenames = aScanner.getIncludedFiles ();
338     if (aXMLFilenames != null)
339     {
340       for (final String sXMLFilename : aXMLFilenames)
341       {
342         final File aXMLFile = new File (aXMLDirectory, sXMLFilename);
343 
344         // Validate XML file
345         getLog ().info ("Validating XML file '" +
346                         aXMLFile.getPath () +
347                         "' against Schematron rules from '" +
348                         schematronFile +
349                         "' expecting " +
350                         (bExpectSuccess ? "success" : "failure"));
351         try
352         {
353           final SchematronOutputType aSOT = aSch.applySchematronValidationToSVRL (TransformSourceFactory.create (aXMLFile));
354 
355           if (aSVRLDirectory != null)
356           {
357             // Save SVRL
358             final File aSVRLFile = new File (aSVRLDirectory, sXMLFilename + ".svrl");
359             if (!aSVRLFile.getParentFile ().mkdirs ())
360               getLog ().error ("Failed to create parent directory of '" + aSVRLFile.getAbsolutePath () + "'!");
361 
362             if (new SVRLMarshaller ().write (aSOT, aSVRLFile).isSuccess ())
363               getLog ().info ("Successfully saved SVRL file '" + aSVRLFile.getPath () + "'");
364             else
365               getLog ().error ("Error saving SVRL file '" + aSVRLFile.getPath () + "'");
366           }
367 
368           final ICommonsList <SVRLFailedAssert> aFailedAsserts = SVRLHelper.getAllFailedAssertions (aSOT);
369           if (bExpectSuccess)
370           {
371             // No failed assertions expected
372             if (aFailedAsserts.isNotEmpty ())
373             {
374               final String sMessage = aFailedAsserts.size () +
375                                       " failed Schematron assertions for XML file '" +
376                                       aXMLFile.getPath () +
377                                       "'";
378               getLog ().error (sMessage);
379               aFailedAsserts.forEach (x -> getLog ().error (x.getAsResourceError (aXMLFile.getPath ())
380                                                              .getAsString (Locale.US)));
381               throw new MojoFailureException (sMessage);
382             }
383           }
384           else
385           {
386             // At least one failed assertions expected
387             if (aFailedAsserts.isEmpty ())
388             {
389               final String sMessage = "No failed Schematron assertions for erroneous XML file '" +
390                                       aXMLFile.getPath () +
391                                       "'";
392               getLog ().error (sMessage);
393               throw new MojoFailureException (sMessage);
394             }
395           }
396         }
397         catch (final MojoFailureException | MojoExecutionException up)
398         {
399           throw up;
400         }
401         catch (final Exception ex)
402         {
403           final String sMessage = "Exception validating XML '" +
404                                   aXMLFile.getPath () +
405                                   "' against Schematron rules from '" +
406                                   schematronFile +
407                                   "'";
408           getLog ().error (sMessage, ex);
409           throw new MojoExecutionException (sMessage, ex);
410         }
411       }
412     }
413   }

415   public void execute () throws MojoExecutionException, MojoFailureException
416   {
417     StaticLoggerBinder.getSingleton ().setMavenLog (getLog ());
418     if (schematronFile == null)
419       throw new MojoExecutionException ("No Schematron file specified!");
420     if (schematronFile.exists () && !schematronFile.isFile ())
421       throw new MojoExecutionException ("The specified Schematron file " + schematronFile + " is not a file!");
422     if (schematronProcessingEngine == null)
423       throw new MojoExecutionException ("An invalid Schematron processing instance is specified! Only one of the following values is allowed: " +
424                                         StringHelper.getImplodedMapped (", ",
425                                                                         ESchematronMode.values (),
426                                                                         x -> "'" + x.getID () + "'"));
427     if (xmlDirectory == null && xmlErrorDirectory == null)
428       throw new MojoExecutionException ("No XML directory specified - positive or negative directory must be present!");
429 
430     if (xmlDirectory != null)
431     {
432       if (xmlDirectory.exists () && !xmlDirectory.isDirectory ())
433         throw new MojoExecutionException ("The specified XML directory " + xmlDirectory + " is not a directory!");
434       if (StringHelper.hasNoText (xmlIncludes))
435         throw new MojoExecutionException ("No XML include pattern specified!");
436 
437       if (svrlDirectory != null)
438       {
439         if (!svrlDirectory.exists () && !svrlDirectory.mkdirs ())
440           throw new MojoExecutionException ("Failed to create the SVRL directory " + svrlDirectory);
441       }
442     }
443 
444     if (xmlErrorDirectory != null)
445     {
446       if (xmlErrorDirectory.exists () && !xmlErrorDirectory.isDirectory ())
447         throw new MojoExecutionException ("The specified erroneous XML directory " +
448                                           xmlErrorDirectory +
449                                           " is not a directory!");
450       if (StringHelper.hasNoText (xmlErrorIncludes))
451         throw new MojoExecutionException ("No erroneous XML include pattern specified!");
452 
453       if (svrlErrorDirectory != null)
454       {
455         if (!svrlErrorDirectory.exists () && !svrlErrorDirectory.mkdirs ())
456           throw new MojoExecutionException ("Failed to create the erroneous SVRL directory " + svrlErrorDirectory);
457       }
458     }
459 
460     // 1. Parse Schematron file
461     final Locale aDisplayLocale = Locale.US;
462     ISchematronResource aSch;
463     IErrorList aSCHErrors;
464     switch (ESchematronMode.getFromIDOrNull (schematronProcessingEngine))
465     {
466       case PURE:
467       {
468         // pure
469         final CollectingPSErrorHandler aErrorHdl = new CollectingPSErrorHandler ();
470         final SchematronResourcePure aRealSCH = new SchematronResourcePure (new FileSystemResource (schematronFile));
471         aRealSCH.setPhase (phaseName);
472         aRealSCH.setErrorHandler (aErrorHdl);
473         aRealSCH.validateCompletely ();
474 
475         aSch = aRealSCH;
476         aSCHErrors = aErrorHdl.getAllErrors ();
477         break;
478       }
479       case SCHEMATRON:
480       {
481         // SCH
482         final CollectingTransformErrorListener aErrorHdl = new CollectingTransformErrorListener ();
483         final SchematronResourceSCH aRealSCH = new SchematronResourceSCH (new FileSystemResource (schematronFile));
484         aRealSCH.setPhase (phaseName);
485         aRealSCH.setLanguageCode (languageCode);
486         aRealSCH.setErrorListener (aErrorHdl);
487         aRealSCH.isValidSchematron ();
488 
489         aSch = aRealSCH;
490         aSCHErrors = aErrorHdl.getErrorList ();
491         break;
492       }
493       case XSLT:
494       {
495         // SCH
496         final CollectingTransformErrorListener aErrorHdl = new CollectingTransformErrorListener ();
497         final SchematronResourceXSLT aRealSCH = new SchematronResourceXSLT (new FileSystemResource (schematronFile));
498         // phase is ignored
499         aRealSCH.setErrorListener (aErrorHdl);
500         aRealSCH.isValidSchematron ();
501 
502         aSch = aRealSCH;
503         aSCHErrors = aErrorHdl.getErrorList ();
504         break;
505       }
506       default:
507         throw new MojoExecutionException ("No handler for processing engine '" + schematronProcessingEngine + "'");
508     }
509     if (aSCHErrors != null)
510     {
511       // Error validating the Schematrons!!
512       boolean bAnyError = false;
513       for (final IError aError : aSCHErrors)
514         if (aError.getErrorLevel ().isGE (EErrorLevel.ERROR))
515         {
516           getLog ().error ("Error in Schematron: " + aError.getAsString (aDisplayLocale));
517           bAnyError = true;
518         }
519         else
520           if (aError.getErrorLevel ().isGE (EErrorLevel.WARN))
521             getLog ().warn ("Warning in Schematron: " + aError.getAsString (aDisplayLocale));
522       if (bAnyError)
523         throw new MojoExecutionException ("The provided Schematron file contains errors. See log for details.");
524     }
525     getLog ().info ("Successfully parsed Schematron file '" + schematronFile.getPath () + "'");
526 
527     // 2. for all XML files that match the pattern
528     if (xmlDirectory != null)
529       _performValidation (aSch, xmlDirectory, xmlIncludes, xmlExcludes, svrlDirectory, true);
530     if (xmlErrorDirectory != null)
531       _performValidation (aSch, xmlErrorDirectory, xmlErrorIncludes, xmlErrorExcludes, svrlErrorDirectory, false);
532   }

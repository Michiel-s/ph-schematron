17  package com.helger.schematron.testfiles;
18  
19  import javax.annotation.Nonnull;
20  
21  import com.helger.commons.ValueEnforcer;
22  import com.helger.commons.annotation.Nonempty;
23  import com.helger.commons.hashcode.HashCodeGenerator;
24  import com.helger.commons.io.resource.IReadableResource;
25  import com.helger.commons.string.ToStringGenerator;
26  
27  /**
28   * Represents a single Schematron test file
29   *
30   * @author Philip Helger
31   */
32  public class SchematronTestFile
33  {
34    private final String m_sParentDirBaseName;
35    private final IReadableResource m_aRes;
36    // For easier usage only:
37    private final String m_sFileBaseName;
38  
39    public SchematronTestFile (@Nonnull @Nonempty final String sParentDirBaseName,
40                               @Nonnull final IReadableResource aRes,
41                               @Nonnull @Nonempty final String sFileBaseName)
42    {
43      ValueEnforcer.notEmpty (sParentDirBaseName, "ParentDirBaseName");
44      ValueEnforcer.notNull (aRes, "Resource");
45      ValueEnforcer.notEmpty (sFileBaseName, "FileBaseName");
46  
47      m_sParentDirBaseName = sParentDirBaseName;
48      m_aRes = aRes;
49      m_sFileBaseName = sFileBaseName;
50    }
51  
52    @Nonnull
53    @Nonempty
54    public String getParentDirBaseName ()
55    {
56      return m_sParentDirBaseName;
57    }
58  
59    @Nonnull
60    public IReadableResource getResource ()
61    {
62      return m_aRes;
63    }
64  
65    @Nonnull
66    @Nonempty
67    public String getFileBaseName ()
68    {
69      return m_sFileBaseName;
70    }
71  
72    @Override
73    public boolean equals (final Object o)
74    {
75      if (o == this)
76        return true;
77      if (o == null || !getClass ().equals (o.getClass ()))
78        return false;
79      final SchematronTestFile rhs = (SchematronTestFile) o;
80      return m_sParentDirBaseName.equals (rhs.m_sParentDirBaseName) && m_aRes.equals (rhs.m_aRes);
81    }
82  
83    @Override
84    public int hashCode ()
85    {
86      return new HashCodeGenerator (this).append (m_sParentDirBaseName).append (m_aRes).getHashCode ();
87    }
88  
89    @Override
90    public String toString ()
91    {
92      return new ToStringGenerator (this).append ("parentDirBaseName", m_sParentDirBaseName)
93                                         .append ("res", m_aRes)
94                                         .append ("fileBaseName", m_sFileBaseName)
95                                         .getToString ();
96    }
97  }

39  /**
40   * ANT task to perform Schematron preprocessing. It converts an existing schema
41   * to the minimal syntax (by default) but allows for a certain degree of
42   * customization by keeping certain elements in the resulting schema. The actual
43   * query binding is used, so that report test expressions can be converted to
44   * assertions, and to replace the content of &lt;param&gt; elements into actual
45   * values.
46   *
47   * @author Philip Helger
48   * @since 5.0.0
49   */
50  public class SchematronPreprocess extends Task
51  {
52    /**
53     * The Schematron source file to be pre-processed.
54     */
55    private File m_aSrcFile;
56  
57    /**
58     * The Schematron destination file to be written.
59     */
60    private File m_aDstFile;
61  
62    /**
63     * <code>true</code> if &lt;title&gt;-elements should be kept.
64     */
65    private boolean m_bKeepTitles = PSPreprocessor.DEFAULT_KEEP_TITLES;
66  
67    /**
68     * <code>true</code> if &lt;diagnostic&gt;-elements should be kept.
69     */
70    private boolean m_bKeepDiagnostics = PSPreprocessor.DEFAULT_KEEP_DIAGNOSTICS;
71  
72    /**
73     * Should &lt;report&gt;-elements be kept or should they be converted to
74     * &lt;assert&gt;-elements?
75     */
76    private boolean m_bKeepReports = PSPreprocessor.DEFAULT_KEEP_REPORTS;
77  
78    /**
79     * Should &lt;pattern&gt;-elements without a single rule be kept or deleted?
80     */
81    private boolean m_bKeepEmptyPatterns = PSPreprocessor.DEFAULT_KEEP_EMPTY_PATTERNS;
82  
83    /**
84     * <code>true</code> if the build should fail if any error occurs. Defaults to
85     * <code>true</code>.
86     */
87    private boolean m_bFailOnError = true;
88  
89    public SchematronPreprocess ()
90    {}
91  
92    public void setSrcFile (@Nonnull final File aFile)
93    {
94      m_aSrcFile = aFile;
95      if (!m_aSrcFile.isAbsolute ())
96        m_aSrcFile = new File (getProject ().getBaseDir (), aFile.getPath ());
97      log ("Using source Schematron file '" + m_aSrcFile + "'", Project.MSG_DEBUG);
98    }
99  
100   public void setDstFile (@Nonnull final File aFile)
101   {
102     m_aDstFile = aFile;
103     if (!m_aDstFile.isAbsolute ())
104       m_aDstFile = new File (getProject ().getBaseDir (), aFile.getPath ());
105     log ("Using destination Schematron file '" + m_aDstFile + "'", Project.MSG_DEBUG);
106   }
107 
108   public void setKeepTitles (final boolean bKeepTitles)
109   {
110     m_bKeepTitles = bKeepTitles;
111     log (bKeepTitles ? "Keeping <title>-elements." : "Removing <title>-elements.", Project.MSG_DEBUG);
112   }
113 
114   public void setKeepDiagnostics (final boolean bKeepDiagnostics)
115   {
116     m_bKeepDiagnostics = bKeepDiagnostics;
117     log (bKeepDiagnostics ? "Keeping <diagnostic>-elements." : "Removing <diagnostic>-elements.", Project.MSG_DEBUG);
118   }
119 
120   public void setKeepReports (final boolean bKeepReports)
121   {
122     m_bKeepReports = bKeepReports;
123     log (bKeepReports ? "Keeping <report>-elements." : "Changing to <assert>-elements.", Project.MSG_DEBUG);
124   }
125 
126   public void setKeepEmptyPatterns (final boolean bKeepEmptyPatterns)
127   {
128     m_bKeepEmptyPatterns = bKeepEmptyPatterns;
129     log (bKeepEmptyPatterns ? "Keeping <pattern>-elements without rules."
130                             : "Deleting <pattern>-elements without rules.",
131          Project.MSG_DEBUG);
132   }
133 
134   public void setFailOnError (final boolean bFailOnError)
135   {
136     m_bFailOnError = bFailOnError;
137 
138     log (bFailOnError ? "Will fail on error" : "Will not fail on error", Project.MSG_DEBUG);
139   }
140 
141   private void _error (@Nonnull final String sMsg)
142   {
143     _error (sMsg, null);
144   }
145 
146   private void _error (@Nonnull final String sMsg, @Nullable final Throwable t)
147   {
148     if (m_bFailOnError)
149       throw new BuildException (sMsg, t);
150     log (sMsg, t, Project.MSG_ERR);
151   }
152 
153   @Override
154   public void execute () throws BuildException
155   {
156     boolean bCanRun = false;
157     if (m_aSrcFile == null)
158       _error ("No source Schematron file specified!");
159     else
160       if (m_aSrcFile.exists () && !m_aSrcFile.isFile ())
161         _error ("The specified source Schematron file " + m_aSrcFile + " is not a file!");
162       else
163         if (m_aDstFile == null)
164           _error ("No destination Schematron file specified!");
165         else
166           if (m_aDstFile.exists () && !m_aDstFile.isFile ())
167             _error ("The specified destination Schematron file " + m_aDstFile + " is not a file!");
168           else
169             bCanRun = true;
170 
171     if (bCanRun)
172       try
173       {
174         // Read source
175         final PSSchema aSchema = new PSReader (new FileSystemResource (m_aSrcFile)).readSchema ();
176 
177         // Setup preprocessor
178         final PSPreprocessor aPreprocessor = new PSPreprocessor (PSXPathQueryBinding.getInstance ());
179         aPreprocessor.setKeepTitles (m_bKeepTitles);
180         aPreprocessor.setKeepDiagnostics (m_bKeepDiagnostics);
181         aPreprocessor.setKeepReports (m_bKeepReports);
182         aPreprocessor.setKeepEmptyPatterns (m_bKeepEmptyPatterns);
183         aPreprocessor.setKeepEmptySchema (true);
184 
185         // Main pre-processing
186         final PSSchema aPreprocessedSchema = aPreprocessor.getAsPreprocessedSchema (aSchema);
187 
188         // Write the result file
189         new PSWriter (new PSWriterSettings ().setXMLWriterSettings (new XMLWriterSettings ())).writeToFile (aPreprocessedSchema,
190                                                                                                             m_aDstFile);
191         log ("Successfully pre-processed Schematron " + m_aSrcFile + " to " + m_aDstFile);
192       }
193       catch (final SchematronReadException | SchematronPreprocessException ex)
194       {
195         _error ("Error processing Schemtron " + m_aSrcFile.getAbsolutePath (), ex);
196       }
197   }
198 }

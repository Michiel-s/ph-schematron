44    public static void validateAndProduceSVRL (@Nonnull final File aSchematron, final File aXML) throws Exception
45    {
46      final PSSchema aSchema = new PSReader (new FileSystemResource (aSchematron)).readSchema ();
47      final PSPreprocessor aPreprocessor = new PSPreprocessor (PSXPathQueryBinding.getInstance ());
48      final PSSchema aPreprocessedSchema = aPreprocessor.getAsPreprocessedSchema (aSchema);
49      final String sSCH = new PSWriter (new PSWriterSettings ().setXMLWriterSettings (new XMLWriterSettings ())).getXMLString (aPreprocessedSchema);
50  
51      if (false)
52        System.out.println (sSCH);
53  
54      final SchematronResourceSCH aSCH = new SchematronResourceSCH (new ReadableResourceString (sSCH,
55                                                                                                StandardCharsets.UTF_8));
56  
57      // Perform validation
58      final SchematronOutputType aSVRL = aSCH.applySchematronValidationToSVRL (new FileSystemResource (aXML));
59      assertNotNull (aSVRL);
60      if (false)
61        System.out.println (new SVRLMarshaller ().getAsString (aSVRL));
62    }
63  
64    @Test
65    public void testIssue () throws Exception
66    {
67      SchematronDebug.setSaveIntermediateXSLTFiles (true);
68  
69      validateAndProduceSVRL (new File ("src/test/resources/issues/github48/schematron.sch"),
70                              new File ("src/test/resources/issues/github48/test.xml"));
71    }
